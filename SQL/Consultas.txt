SELECT CPF,COLABORADORES.NOME,DEPARTAMENTOS.NOME,filiais.cidade
FROM COLABORADORES
INNER JOIN DEPARTAMENTOS 
ON COLABORADORES.IDDEPARTAMENTO = DEPARTAMENTOS.IDDEPARTAMENTO
INNER JOIN FILIAIS
ON (COLABORADORES.IDDEPARTAMENTO = DEPARTAMENTOS.IDDEPARTAMENTO)
WHERE DEPARTAMENTOS.idfilial = filiais.idfilial
ORDER BY filiais.idfilial,colaboradores.nome ASC;



SELECT CPF,COLABORADORES.NOME,DATAADIMISSAO,FILIAIS.CIDADE
FROM COLABORADORES
INNER JOIN DEPARTAMENTOS
ON COLABORADORES.IDDEPARTAMENTO = DEPARTAMENTOS.IDDEPARTAMENTO
INNER JOIN FILIAIS
ON (COLABORADORES.IDDEPARTAMENTO = DEPARTAMENTOS.IDDEPARTAMENTO)
WHERE DEPARTAMENTOS.IDFILIAL = FILIAIS.IDFILIAL
ORDER BY DATAADIMISSAO ASC
LIMIT 5;



SELECT FILIAIS.IDFILIAL,CIDADE,COUNT(DISTINCT COLABORADORES.CPF) AS QUANTIACOLABORADOR
FROM FILIAIS
INNER JOIN DEPARTAMENTOS
ON (FILIAIS.IDFILIAL = DEPARTAMENTOS.IDFILIAL)
INNER JOIN COLABORADORES
ON DEPARTAMENTOS.IDDEPARTAMENTO = COLABORADORES.IDDEPARTAMENTO
GROUP BY FILIAIS.IDFILIAL;


SELECT DEPARTAMENTOS.IDDEPARTAMENTO,DEPARTAMENTOS.NOME,AVG(COLABORADORES.SALARIO)
FROM DEPARTAMENTOS
INNER JOIN COLABORADORES
ON DEPARTAMENTOS.IDDEPARTAMENTO = COLABORADORES.IDDEPARTAMENTO
GROUP BY DEPARTAMENTOS.IDDEPARTAMENTO;
